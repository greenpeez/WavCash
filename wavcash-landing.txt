precision highp float;
uniform sampler2D src;
uniform vec2 center;

  vec4 s = texture(src, uv);
  float n = texture(src, uv + vec2(0, texel.y)).r;
  float south = texture(src, uv - vec2(0, texel.y)).r;
  float e = texture(src, uv + vec2(texel.x, 0)).r;
  float w = texture(src, uv - vec2(texel.x, 0)).r;
  float avg = (n + south + e + w) * 0.25;
  s.g += (avg - s.r) * 0.18;
  s.g *= 0.945;
  s.r += s.g;
  s.r *= 0.9992;
  // Absorb at edges
  vec2 a = smoothstep(vec2(0), vec2(0.01), uv);
  vec2 b = smoothstep(vec2(0), vec2(0.01), 1.0 - uv);
  float edge = a.x * a.y * b.x * b.y;
  s.rg *= edge;
  dst = s;
}`;

// ---- RENDER: heightfield -> mercury with HARD BRIGHTNESS LIMITER ----
const FRAG_RENDER = `#version 300 es
precision highp float;
uniform sampler2D hmap;
uniform vec2 texel;
uniform vec2 res;
uniform float time;
uniform int uLightMode;
in vec2 uv;
out vec4 fc;

vec3 getNorm(vec2 p) {
  float l = texture(hmap, p - vec2(texel.x, 0)).r;
  float r = texture(hmap, p + vec2(texel.x, 0)).r;
  float u = texture(hmap, p + vec2(0, texel.y)).r;
  float d = texture(hmap, p - vec2(0, texel.y)).r;
  float ns = uLightMode == 1 ? 2.0 : 1.0;
  return normalize(vec3((l - r) * ns, (d - u) * ns, 1.0));
}

vec3 env(vec3 rd) {
  float y = rd.y;

  vec3 dark, mid, tint;
  float limiter;

  if (uLightMode == 1) {
    dark = vec3(0.62, 0.62, 0.62);
    mid  = vec3(0.78, 0.78, 0.77);
    tint = vec3(0.92, 0.91, 0.89);
    limiter = 0.93;
  } else {
    dark = vec3(0.01, 0.01, 0.01);
    mid  = vec3(0.08, 0.08, 0.07);
    tint = vec3(0.25, 0.24, 0.22);
    limiter = 0.12;
  }

  vec3 c = mix(dark, mid, smoothstep(-0.5, 1.0, y));

  // Soft light bands
  float b1 = smoothstep(0.2, 0.35, y) * (1.0 - smoothstep(0.55, 0.65, y));
  c += tint * 0.25 * b1;

  float b2 = smoothstep(0.7, 0.78, y) * (1.0 - smoothstep(0.85, 0.92, y));
  c += tint * 0.15 * b2;

  // Low fill
  c += dark * 1.5 * smoothstep(-0.8, -0.3, y) * (1.0 - smoothstep(-0.05, 0.15, y));

  // Side variation
  c *= 0.90 + 0.10 * (sin(rd.x * 3.0 + rd.z * 1.5) * 0.5 + 0.5);

  // >>> HARD BRIGHTNESS LIMITER <<<
  c = min(c, vec3(limiter));

  return c;
}

void main() {
  vec3 N = getNorm(uv);
  float asp = res.x / res.y;
  vec3 V = normalize(vec3((uv - 0.5) * vec2(asp, 1.0) * 0.3, -1.0));
  vec3 R = reflect(V, N);

  float NdotV = max(dot(N, -V), 0.0);
  // Fresnel — brighter in light mode for chrome look
  float fLo = uLightMode == 1 ? 0.75 : 0.65;
  float fHi = uLightMode == 1 ? 0.90 : 0.75;
  float fresnel = mix(fLo, fHi, 1.0 - NdotV);

  vec3 c = env(R) * fresnel;

  // ACES tonemap
  c = (c * (2.51 * c + 0.03)) / (c * (2.43 * c + 0.59) + 0.14);
  // Gamma
  c = pow(clamp(c, 0.0, 1.0), vec3(1.0 / 2.2));

  // >>> FINAL OUTPUT LIMITER (dark mode only) <<<
  if (uLightMode == 0) {
    c = min(c, vec3(0.20));
  }

  fc = vec4(c, 1);
}`;

function mkShader(src, type) {
  const s = gl.createShader(type);
  gl.shaderSource(s, src);
  gl.compileShader(s);
  if (!gl.getShaderParameter(s, gl.COMPILE_STATUS)) {
    L('SHADER ERR: ' + gl.getShaderInfoLog(s));
    return null;
  }
  return s;
}

function mkProg(vs, fs, name) {
  const v = mkShader(vs, gl.VERTEX_SHADER);
  const f = mkShader(fs, gl.FRAGMENT_SHADER);
  if (!v || !f) { L(name + ': compile FAIL'); return null; }
  const p = gl.createProgram();
  gl.attachShader(p, v);
  gl.attachShader(p, f);
  gl.linkProgram(p);
  if (!gl.getProgramParameter(p, gl.LINK_STATUS)) {
    L(name + ' link FAIL: ' + gl.getProgramInfoLog(p));
    return null;
  }
  L(name + ': OK');
  return p;
}

function mkTex() {
  const t = gl.createTexture();
  gl.bindTexture(gl.TEXTURE_2D, t);
  gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA32F, SIM, SIM, 0, gl.RGBA, gl.FLOAT, null);
  gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
  gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);
  gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
  gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
  return t;
}

function mkFBO(t, label) {
  const f = gl.createFramebuffer();
  gl.bindFramebuffer(gl.FRAMEBUFFER, f);
  gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, t, 0);
  const s = gl.checkFramebufferStatus(gl.FRAMEBUFFER);
  if (s !== gl.FRAMEBUFFER_COMPLETE) {
    L(label + ' FBO: FAIL 0x' + s.toString(16));
  } else {
    L(label + ' FBO: OK');
  }
  gl.bindFramebuffer(gl.FRAMEBUFFER, null);
  return f;
}

function runDrop(cx, cy, rad, str) {

}

  gl.bindVertexArray(vao);
  gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
  swap();
}

// ============================================================
// RENDER LOOP — continuous cursor tracking restored
// ============================================================

const t0 = performance.now();
let frame = 0;

function loop() {
  requestAnimationFrame(loop);
  const now = performance.now();
  const time = (now - t0) * 0.001;
  frame++;

  // ---- Continuous cursor tracking ----
  const dx = mx - pmx, dy = my - pmy;
  const vel = Math.sqrt(dx * dx + dy * dy) * 60;
  pmx = mx; pmy = my;

  const moving = (now - lastMove) < 150;
  const rawStr = moving ? 0.15 + Math.min(vel, 3.0) * 0.15 : 0.0;

  if (rawStr >= smoothStr) {
    smoothStr = rawStr; // instant rise
  } else {
    smoothStr += (rawStr - smoothStr) * 0.04; // slow fall
  }

  if (smoothStr > 0.001) {
    runDrop(mx, my, 0.015, smoothStr);
  }

  // ---- Update wave sim ----
  runUpdate();

  // ---- Ambient ripples when idle ----
  if (!moving && frame % 90 === 0) {
    const ax = 0.15 + Math.random() * 0.7, ay = 0.15 + Math.random() * 0.7;
    runDrop(ax, ay, 0.035, 0.04);
  }

  // ---- Render to screen ----
  gl.bindFramebuffer(gl.FRAMEBUFFER, null);
  gl.viewport(0, 0, canvas.width, canvas.height);

  if (mode === 0) {
    gl.useProgram(pDebug);
    gl.activeTexture(gl.TEXTURE0);
    gl.bindTexture(gl.TEXTURE_2D, tex[cR]);
    gl.uniform1i(uDebug.hmap, 0);
  } else {
    gl.useProgram(pRender);
    gl.activeTexture(gl.TEXTURE0);
    gl.bindTexture(gl.TEXTURE_2D, tex[cR]);
    gl.uniform1i(uRender.hmap, 0);
    gl.uniform2f(uRender.texel, 1.0/SIM, 1.0/SIM);
    gl.uniform2f(uRender.res, canvas.width, canvas.height);
    gl.uniform1f(uRender.time, time);
    gl.uniform1i(uRender.uLightMode, isLightMode ? 1 : 0);
  }

  gl.bindVertexArray(vao);
  gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
}

L('Starting render loop...');
requestAnimationFrame(loop);

function animateCounter(el, target, prefix, suffix, duration) {
  const start = performance.now();
  const isFloat = target % 1 !== 0;
  function tick(now) {
    const t = Math.min((now - start) / duration, 1);
    const ease = 1 - Math.pow(1 - t, 3);
    const val = target * ease;
    if (isFloat) {
      el.textContent = prefix + val.toFixed(1) + suffix;
    } else {
      el.textContent = prefix + Math.round(val).toLocaleString() + suffix;
    }
    if (t < 1) requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
}

// ============================================================
// AMBER ACCENT: ignite words + Apple-style outline flare
// ============================================================
(function() {
  function lerp(a, b, t) { return Math.round(a + (b - a) * t); }
  function easeIO(t) { return t < 0.5 ? 2*t*t : 1 - Math.pow(-2*t+2, 2)/2; }

  // ---- Apple cutout flare: SVG text stroke with animated dash offset ----
  function createFlare(el) {
    const text = el.textContent;
    const rect = el.getBoundingClientRect();
    const h1 = el.closest('h1');
    const cs = getComputedStyle(h1);
    // Get the actual rendered font size from the span itself
    const spanCS = getComputedStyle(el);
    const fontSize = parseFloat(spanCS.fontSize);
    const fontFamily = cs.fontFamily;
    const fontWeight = cs.fontWeight;
    const letterSpacing = cs.letterSpacing;

    // Create SVG overlay
    const ns = 'http://www.w3.org/2000/svg';
    const svg = document.createElementNS(ns, 'svg');
    svg.setAttribute('class', 'flare-svg');
    svg.setAttribute('viewBox', `0 0 ${rect.width} ${rect.height}`);
    svg.setAttribute('xmlns', ns);

    // Glow filter for the flare point
    const defs = document.createElementNS(ns, 'defs');
    const filter = document.createElementNS(ns, 'filter');
    filter.setAttribute('id', 'flare-glow-' + text.replace(/[^a-z]/gi, ''));
    filter.setAttribute('x', '-50%'); filter.setAttribute('y', '-50%');
    filter.setAttribute('width', '200%'); filter.setAttribute('height', '200%');
    const blur = document.createElementNS(ns, 'feGaussianBlur');
    blur.setAttribute('stdDeviation', '3');
    blur.setAttribute('result', 'blur');
    const merge = document.createElementNS(ns, 'feMerge');
    const m1 = document.createElementNS(ns, 'feMergeNode'); m1.setAttribute('in', 'blur');
    const m2 = document.createElementNS(ns, 'feMergeNode'); m2.setAttribute('in', 'SourceGraphic');
    merge.appendChild(m1); merge.appendChild(m2);
    filter.appendChild(blur); filter.appendChild(merge);
    defs.appendChild(filter);
    svg.appendChild(defs);

    // Text element — stroke only, no fill
    const svgText = document.createElementNS(ns, 'text');
    svgText.setAttribute('x', '0');
    // Baseline: place text so it aligns with DOM text. ~75% of font size is a good baseline approx.
    svgText.setAttribute('y', String(fontSize * 0.82));
    svgText.setAttribute('font-family', fontFamily);
    svgText.setAttribute('font-weight', fontWeight);
    svgText.setAttribute('font-size', fontSize + 'px');
    if (letterSpacing && letterSpacing !== 'normal') {
      svgText.setAttribute('letter-spacing', letterSpacing);
    }
    svgText.setAttribute('fill', 'none');
    svgText.setAttribute('stroke', getFlareColor(0.9));
    svgText.setAttribute('stroke-width', '0.8');
    svgText.setAttribute('filter', `url(#flare-glow-${text.replace(/[^a-z]/gi, '')})`);
    svgText.textContent = text;
    svg.appendChild(svgText);

    el.appendChild(svg);

    // Measure the total path length of the text stroke
    // SVGTextElement doesn't have getTotalLength, so we estimate from character count + font size
    // A reasonable estimate: each glyph outline ≈ 3× font size in path length
    const estimatedLength = text.length * fontSize * 3;

    // Dash pattern: short bright segment + long gap
    const dashLen = estimatedLength * 0.08; // 8% of outline is lit
    const gapLen = estimatedLength - dashLen;
    svgText.setAttribute('stroke-dasharray', `${dashLen} ${gapLen}`);

    // Start fully hidden (offset = full length)
    svgText.setAttribute('stroke-dashoffset', String(estimatedLength));
    svgText.style.opacity = '0';

    return { svg, svgText, totalLength: estimatedLength, dashLen, gapLen };
  }

  // Animate the flare — continuous loop
  const flareTargets = [];
  function flareLoop(now) {
    requestAnimationFrame(flareLoop);
    for (const f of flareTargets) {
      // 2.5s trace + 7.5s gap = 10s total cycle
      const elapsed = now - f.startTime;
      const cycle = 10000;
      const pos = elapsed % cycle;
      const traceTime = 2500;

      if (pos < traceTime) {
        // Active trace phase: sweep across the outline in 2.5s
        const phase = pos / traceTime;
        const offset = f.totalLength - (phase * f.totalLength * 2);
        f.svgText.setAttribute('stroke-dashoffset', String(offset));
        // Fade in at start, fade out at end of trace
        const fade = phase < 0.1 ? phase / 0.1
                   : phase > 0.85 ? (1 - phase) / 0.15
                   : 1.0;
        f.svgText.setAttribute('stroke', getFlareColor((fade * 0.9).toFixed(2)));
      } else {
        // Gap phase: hidden
        f.svgText.setAttribute('stroke', getFlareColor(0));
      }
    }
  }
  requestAnimationFrame(flareLoop);

  // ---- Ignite a word: color interpolation, then start flare after optional delay ----
  function igniteWord(selector, delay, flareDelay) {
    flareDelay = flareDelay || 0;
    setTimeout(() => {
      const el = document.querySelector(selector);
      if (!el) return;
      const from = getHoverBase('h1');
      const to = [[212,136,58], [212,136,58], [212,136,58]];
      const dur = 2000;
      const start = performance.now();
      el.style.animation = 'none';
      el.style.backgroundSize = '100% 100%';

      // Pre-create the flare SVG (hidden until flare engages)
      const flare = createFlare(el);

      function startFlare() {
        flare.svgText.style.opacity = '1';
        flare.svgText.style.transition = 'opacity 0.6s ease';
        flareTargets.push({
          svgText: flare.svgText,
          totalLength: flare.totalLength,
          startTime: performance.now()
        });
      }

      function tick(now) {
        const t = Math.min((now - start) / dur, 1);
        const e = easeIO(t);
        const c0 = `rgb(${lerp(from[0][0],to[0][0],e)},${lerp(from[0][1],to[0][1],e)},${lerp(from[0][2],to[0][2],e)})`;
        const c1 = `rgb(${lerp(from[1][0],to[1][0],e)},${lerp(from[1][1],to[1][1],e)},${lerp(from[1][2],to[1][2],e)})`;
        const c2 = `rgb(${lerp(from[2][0],to[2][0],e)},${lerp(from[2][1],to[2][1],e)},${lerp(from[2][2],to[2][2],e)})`;
        el.style.background = `linear-gradient(135deg, ${c0} 0%, ${c1} 40%, ${c2} 100%)`;
        el.style.webkitBackgroundClip = 'text';
        el.style.backgroundClip = 'text';
        if (t < 1) {
          requestAnimationFrame(tick);
        } else {
          // Lock to flat amber
          el.style.background = '#D4883A';
          el.style.webkitBackgroundClip = 'text';
          el.style.backgroundClip = 'text';
          el.style.webkitTextFillColor = 'transparent';
          // Start flare after optional delay
          if (flareDelay > 0) {
            setTimeout(startFlare, flareDelay);
          } else {
            startFlare();
          }
        }
      }
      requestAnimationFrame(tick);
    }, delay);
  }

  // Timeline:
  // 1.0s — royalties color shift starts
  // 3.5s — royalties color done (locks amber)
  // 4.5s — visible color shift starts (3.5s gap)
  // 7.0s — visible color done (locks amber) → royalties flare fires (3.5s after its color finished)
  // 7.3s — visible flare fires (0.3s after its color finished)
  igniteWord('.word-royalties', 1000, 3500);
  igniteWord('.word-visible', 4500, 300);
})();

// ============================================================
// GRADIENT HOVER — JS interpolation (1s) for smooth amber on any text
// ============================================================
(function() {
  const amber = [212, 136, 58];
  const DUR = 750;
  function lerp(a, b, t) { return Math.round(a + (b - a) * t); }
  function easeInOut(t) { return t < 0.5 ? 2*t*t : 1 - Math.pow(-2*t+2, 2)/2; }

  function attachHover(selector, baseType) {
    document.querySelectorAll(selector).forEach(span => {
      let raf = null, progress = 0, direction = 0, lastTime = 0;

      function paint(t) {
        const base = getHoverBase(baseType);
        const e = easeInOut(t);
        const c0 = `rgb(${lerp(base[0][0],amber[0],e)},${lerp(base[0][1],amber[1],e)},${lerp(base[0][2],amber[2],e)})`;
        const c1 = `rgb(${lerp(base[1][0],amber[0],e)},${lerp(base[1][1],amber[1],e)},${lerp(base[1][2],amber[2],e)})`;
        const c2 = `rgb(${lerp(base[2][0],amber[0],e)},${lerp(base[2][1],amber[1],e)},${lerp(base[2][2],amber[2],e)})`;
        span.style.background = `linear-gradient(135deg, ${c0} 0%, ${c1} 40%, ${c2} 100%)`;
        span.style.backgroundSize = '100% 100%';
        span.style.webkitBackgroundClip = 'text';
        span.style.backgroundClip = 'text';
        span.style.webkitTextFillColor = 'transparent';
      }

      function animate(now) {
        if (!lastTime) lastTime = now;
        const dt = now - lastTime;
        lastTime = now;
        progress += direction * (dt / DUR);
        progress = Math.max(0, Math.min(1, progress));
        paint(progress);
        if ((direction > 0 && progress < 1) || (direction < 0 && progress > 0)) {
          raf = requestAnimationFrame(animate);
        } else {
          raf = null; lastTime = 0;
          if (progress <= 0) {
            span.style.background = '';
            span.style.backgroundSize = '';
            span.style.webkitBackgroundClip = '';
            span.style.backgroundClip = '';
            span.style.webkitTextFillColor = '';
          }
        }
      }

      span.addEventListener('mouseenter', () => {
        direction = 1;
        span.style.animation = 'none';
        if (!raf) { lastTime = 0; raf = requestAnimationFrame(animate); }
      });
      span.addEventListener('mouseleave', () => {
        direction = -1;
        if (!raf) { lastTime = 0; raf = requestAnimationFrame(animate); }
      });
    });
  }

  // H1 words (gradient → amber) — exclude ignited words
  attachHover('.hero h1 .hw:not(.word-royalties):not(.word-visible)', 'h1');
  // P words (muted grey gradient → amber)
  attachHover('.hero p .hw', 'p');
})();

// ATOM CURSOR — spinning electrons on tilted elliptical orbits

(function() {
  const cursor = document.getElementById('atom-cursor');
  if (!cursor) return;
  const electrons = cursor.querySelectorAll('.electron');
  let curX = -100, curY = -100; // start offscreen
  let targetX = -100, targetY = -100;

  // Track mouse
  document.addEventListener('mousemove', e => {
    targetX = e.clientX;
    targetY = e.clientY;
  });
  document.addEventListener('touchmove', e => {
    targetX = e.touches[0].clientX;
    targetY = e.touches[0].clientY;
  }, { passive: true });

  // Hide on buttons, show on everything else
  // Use event delegation for performance
  const hideSelectors = 'button, .nav-cta, .btn-primary, .btn-secondary, .nav-link, input, textarea, select';
  document.addEventListener('mouseover', e => {
    if (e.target.closest(hideSelectors)) {
      cursor.classList.add('hidden');
    }
  });
  document.addEventListener('mouseout', e => {
    if (e.target.closest(hideSelectors)) {
      cursor.classList.remove('hidden');
    }
  });

  // Hide when mouse leaves window
  document.addEventListener('mouseleave', () => { cursor.classList.add('hidden'); });
  document.addEventListener('mouseenter', () => { cursor.classList.remove('hidden'); });

  // Orbit parameters for 3 electrons — tilted ellipses at 60° intervals
  const orbits = [
    { rx: 14, ry: 5, tilt: 0,   speed: 1.8 },   // horizontal orbit
    { rx: 14, ry: 5, tilt: 60,  speed: 2.3 },   // tilted 60°
    { rx: 14, ry: 5, tilt: 120, speed: 1.5 },   // tilted 120°
  ];
  const cx = 18, cy = 18; // SVG center

  function animateCursor(now) {
    requestAnimationFrame(animateCursor);
    const t = now * 0.001; // seconds

    // Smooth follow (slight lag for organic feel)
    curX += (targetX - curX) * 0.25;
    curY += (targetY - curY) * 0.25;

    // Position cursor centered on mouse
    cursor.style.transform = `translate(${curX - 18}px, ${curY - 18}px)`;

    // Animate each electron along its tilted elliptical orbit
    for (let i = 0; i < 3; i++) {
      const o = orbits[i];
      const angle = t * o.speed + (i * Math.PI * 2 / 3); // phase offset
      // Position on untilted ellipse
      const ex = Math.cos(angle) * o.rx;
      const ey = Math.sin(angle) * o.ry;
      // Rotate by tilt
      const rad = o.tilt * Math.PI / 180;
      const rx = ex * Math.cos(rad) - ey * Math.sin(rad);
      const ry = ex * Math.sin(rad) + ey * Math.cos(rad);
      electrons[i].setAttribute('cx', cx + rx);
      electrons[i].setAttribute('cy', cy + ry);
      // Fade when "behind" the nucleus (ey < 0 = behind, based on untilted y)
      const depth = Math.sin(angle); // -1 to 1
      const op = 0.35 + depth * 0.55; // 0.35 when behind, 0.9 when in front
      electrons[i].setAttribute('opacity', op.toFixed(2));
    }
  }
  requestAnimationFrame(animateCursor);
})();

// LIGHT/DARK MODE TOGGLE
// ============================================================
function toggleMode() {
  isLightMode = !isLightMode;
  document.documentElement.classList.toggle('light-mode', isLightMode);

  // Update atom cursor colors
  const cc = isLightMode ? '#1A1A1A' : '#D4883A';
  document.querySelectorAll('#atom-cursor .orbit-path').forEach(el => el.setAttribute('stroke', cc));
  document.querySelectorAll('#atom-cursor .electron').forEach(el => el.setAttribute('fill', cc));
  document.querySelectorAll('#atom-cursor circle:not(.electron)').forEach(el => el.setAttribute('fill', cc));
}

// Toggle button click
document.querySelector('.theme-toggle').addEventListener('click', toggleMode);
